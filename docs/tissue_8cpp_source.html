107 bool Tissue::grow(const size_t max_size) {HERE;
108  init();
109  bool success = false;
110  const size_t max_time = static_cast<size_t>(std::log2(max_size) * 10);
111  size_t i = 0;
112  while (true) {
113  if (tumor_.size() < RECORDING_EARLY_GROWTH_) {
114  write(snapshots_);
115  } else {
116  RECORDING_EARLY_GROWTH_ = 0; // prevent restart by cell death
117  }
118  if (tumor_.size() >= max_size) {success = true; break;}
119  if ((++i % 256) == 0) {DCERR("\r" << tumor_.size());}
120  auto it = queue_.begin();
121  time_ = it->first;
122  if (time_ > max_time) {success = true; break;}
123  const auto mother = it->second;
124  queue_.erase(it);
125  if (mother->next_event() == Event::birth) {
126  const auto daughter = std::make_shared<Cell>(*mother);
127  if (insert(daughter)) {
128  mother->set_time_of_death(time_);
129  write(specimens_, *mother);
130  mother->set_time_of_death(0.0);
131  mother->set_time_of_birth(time_, ++id_tail_);
132  daughter->set_time_of_birth(time_, ++id_tail_);
133  drivers_ << mother->mutate();
134  drivers_ << daughter->mutate();
135  if (tumor_.size() > MUTATION_TIMING_) { // once
136  MUTATION_TIMING_ = std::numeric_limits<size_t>::max();
137  drivers_ << daughter->force_mutate();
138  }
139  queue_push(mother);
140  queue_push(daughter);
141  } else {
142  if (mother->frustration() > 256) break;
143  queue_push(mother);
144  continue; // skip write()
145  }
146  } else if (mother->next_event() == Event::death) {
147  mother->set_time_of_death(time_);
148  write(specimens_, *mother);
149  tumor_.erase(mother);
150  if (tumor_.empty()) break;
151  } else {
152  migrate(mother);
153  queue_push(mother);
154  }
155  }
156  DCERR(std::endl);
157  write(specimens_);
158  return success;
159 }

336 double Tissue::positional_value(const std::valarray<int>& coord) const {
337  if ((SIGMA_E_ > 1e9) | (tumor_.size() <= 8)) return 1.0;
