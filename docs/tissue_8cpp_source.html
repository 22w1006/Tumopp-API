108 bool Tissue::grow(const size_t max_size, const double plateau_time) {HERE;
109  init();
110  bool success = false;
111  double time_stopped = 0.0;
112  const size_t max_time = static_cast<size_t>(std::log2(max_size) * 10);
113  size_t i = 0;
114  while (true) {
115  if (tumor_.size() < RECORDING_EARLY_GROWTH_) {
116  write(snapshots_);
117  } else {
118  RECORDING_EARLY_GROWTH_ = 0; // prevent restart by cell death
119  }
120  if (tumor_.size() >= max_size && time_stopped == 0.0) {
121  // only the first time that tumor_.size() has reached max_size
122  success = true;
123  time_stopped = time_;
124  queue_.clear();
125  for (auto& p: tumor_) {
126  p->increase_death_rate();
127  queue_push(p);
128  }
129  }
130  if ((++i % 256U) == 0U) {DCERR("\r" << tumor_.size());}
131  auto it = queue_.begin();
132  time_ = it->first;
133  if (time_stopped > 0.0 && time_ - time_stopped > plateau_time) {break;}
134  if (time_ > max_time) {success = true; break;}
135  const auto mother = it->second;
136  queue_.erase(it);
137  if (mother->next_event() == Event::birth) {
138  const auto daughter = std::make_shared<Cell>(*mother);
139  if (insert(daughter)) {
140  mother->set_time_of_death(time_);
141  write(specimens_, *mother);
142  mother->set_time_of_death(0.0);
143  mother->set_time_of_birth(time_, ++id_tail_);
144  daughter->set_time_of_birth(time_, ++id_tail_);
145  drivers_ << mother->mutate();
146  drivers_ << daughter->mutate();
147  if (tumor_.size() > MUTATION_TIMING_) { // once
148  MUTATION_TIMING_ = std::numeric_limits<size_t>::max();
149  drivers_ << daughter->force_mutate();
150  }
151  queue_push(mother);
152  queue_push(daughter);
153  } else {
154  if (mother->frustration() > 250U) break;
155  queue_push(mother);
156  continue; // skip write()
157  }
158  } else if (mother->next_event() == Event::death) {
159  mother->set_time_of_death(time_);
160  write(specimens_, *mother);
161  tumor_.erase(mother);
162  if (tumor_.empty()) break;
163  } else {
164  migrate(mother);
165  queue_push(mother);
166  }
167  }
168  DCERR(std::endl);
169  write(specimens_);
170  return success;
171 }

370  const size_t sample_size = samples.size();
371  const auto mutants = generate_neutral_mutations();

386  if (s > 0U) {
387  segsites = wtl::transpose(segsites);

390  for (const auto& x: segsites) {

392  }
