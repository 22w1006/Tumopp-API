157 double Cell::delta_time(const double positional_value) {
158  double t_birth = std::numeric_limits<double>::infinity();
159  double t_death = std::numeric_limits<double>::infinity();
160  double t_migra = std::numeric_limits<double>::infinity();
161  if (proliferation_capacity_ > 0) {
162  double mu = 1.0;
163  mu /= birth_rate_;
164  mu /= positional_value;
165  mu -= elapsed_;
166  double theta = std::max(mu / GAMMA_SHAPE_, 0.0);
167  std::gamma_distribution<double> gamma(GAMMA_SHAPE_, theta);
168  t_birth = gamma(wtl::sfmt());
169  }
170  if (death_rate_ > 0.0) {
171  std::exponential_distribution<double> exponential(death_rate_);
172  t_death = exponential(wtl::sfmt());
173  }
174  if (migra_rate_ > 0.0) {
175  std::exponential_distribution<double> exponential(migra_rate_);
176  t_migra = exponential(wtl::sfmt());
177  }
178 
179  if (t_birth < t_death && t_birth < t_migra) {
180  std::bernoulli_distribution bern_death(death_prob_);
181  if (bern_death(wtl::sfmt())) {
182  next_event_ = Event::death;
183  } else {
184  next_event_ = Event::birth;
185  }
186  elapsed_ = 0.0;
187  return t_birth;
188  } else if (t_death < t_migra) {
189  next_event_ = Event::death;
190  return t_death;
191  } else {
192  next_event_ = Event::migration;
193  elapsed_ += t_migra;
194  return t_migra;
195  }
196 }
